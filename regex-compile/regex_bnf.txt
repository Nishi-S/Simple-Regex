<pattern> ::= <character> <unary-operator>? <pattern>
            | '(' <pattern> ')' <unary-operator>? <pattern>
            | ε

<unary-operator> ::= '*'
                   | '+'
                   | '?'

<character> ::= <escaped-character>
              | <ranged-character>
              | <ascii-character-without-operator>

<escaped-character> ::= '\' ( 'd' | 'w' | 's' )
<ranged-character> ::= '[' <character>+ ']'
